			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:flag_restored_varangian_guard

		}
		title:d_varangian_guard = {
			NOT = {
				has_variable = v_guard
			}
		}
		OR = {
			has_title = title:e_byzantium
			has_title = title:e_roman_empire
		}
	}

		is_independent_ruler = yes
		prestige_level >= high_prestige_level

		is_independent_ruler = yes
		is_at_war = no

			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:flag_restored_varangian_guard
			}
		}
		create_character = {
            culture = culture:norse
            gender = male
            dynasty = generate
            faith = faith:norse_pagan

			trait = varangian
			trait = berserker
			trait = brave

		}

		title:d_varangian_guard = {
			change_title_holder = {
				holder = scope:varangian_guard_holder
				change = scope:title_change
			}
		}
		resolve_title_and_vassal_change = scope:title_change

			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
		hidden_effect = {
			scope:varangian_guard_holder = {
				change_liege = {
					liege = root
					change = scope:change
				}

			}
		}
		resolve_title_and_vassal_change = scope:change

		title:d_varangian_guard = {
			set_variable = {
				name = v_guard
				value = yes
			}
		}
		hidden_effect = {
            if = {
                limit = {
                    is_employed_varangian_captain_court_position = no

	}
